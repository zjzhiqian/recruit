CREATE TABLE `super_merchants` (
  `id` int(11) unsigned NOT NULL AUTO_INCREMENT,
  `merchant_name` varchar(80) NOT NULL DEFAULT '' COMMENT '商家名	',
  `password` varchar(80) NOT NULL DEFAULT '' COMMENT '密码',
  `phone` varchar(20) NOT NULL DEFAULT '' COMMENT '联系电话	',
  `contacts` varchar(80) NOT NULL DEFAULT '' COMMENT '联系人',
  `address` varchar(255) NOT NULL DEFAULT '',
  `recovery_rate` decimal(4,4) NOT NULL DEFAULT '0.0000' COMMENT '留言回复率	',
  `default_reply` text,
  `created_at` timestamp NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `updated_at` timestamp NULL DEFAULT NULL COMMENT '上次更新时间	',
  `deleted_at` timestamp NULL DEFAULT NULL COMMENT '删除时间	',
  `receiver_name` varchar(80) DEFAULT NULL COMMENT '支付宝名称',
  `alipay_account` varchar(80) DEFAULT NULL COMMENT '支付宝账号',
  `roles` varchar(160) DEFAULT 'supplier' COMMENT '用户所属角色(供应商supplier,运营部管理员admin) ',
  PRIMARY KEY (`id`),
  KEY `IDX_MN` (`merchant_name`)
) ENGINE=InnoDB AUTO_INCREMENT=29 DEFAULT CHARSET=utf8;



CREATE TABLE `super_stores` (
  `id` int(11) unsigned NOT NULL AUTO_INCREMENT COMMENT '主键自增',
  `store_name` varchar(80) NOT NULL DEFAULT '' COMMENT '店铺名',
  `notice` varchar(255) DEFAULT NULL COMMENT '店铺公告',
  `logo_url` varchar(255) DEFAULT NULL COMMENT '店铺头像地址',
  `store_banner` varchar(255) DEFAULT NULL COMMENT '店铺背景',
  `merchant_id` int(11) unsigned NOT NULL COMMENT '商家ID merchant表外键	',
  `likes_num` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '获赞次数',
  `is_open` tinyint(1) NOT NULL DEFAULT '0' COMMENT '是否开店 0关 1开	',
  `description` mediumtext COMMENT '店铺宣言',
  `store_link` varchar(255) DEFAULT NULL COMMENT '店铺链接	',
  `created_at` timestamp NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `updated_at` timestamp NULL DEFAULT NULL COMMENT '上级更新时间',
  `deleted_at` timestamp NULL DEFAULT NULL COMMENT '删除时间',
  `rank` int(11) DEFAULT NULL COMMENT '店铺展示顺序',
  `type` int(2) DEFAULT NULL COMMENT '店铺类型（1：本月之星，2：网红店铺，3：学生超级店）',
  `live_address` varchar(255) DEFAULT NULL COMMENT '直播地址',
  `detail_logo` varchar(255) DEFAULT NULL COMMENT '详情Logo',
  PRIMARY KEY (`id`),
  KEY `IDX_MI` (`merchant_id`)
) ENGINE=InnoDB AUTO_INCREMENT=28 DEFAULT CHARSET=utf8;
